AmplifyAuthenticationService({
  @required AuthCategory authServiceInterface,
  @required AmplifyHub authMessageBus,
}) {
  if (authServiceInterface == null || authMessageBus == null) {
    throw BadAuthenticationInterfaceFailure();
  }
  _authServiceInterface = authServiceInterface;
  _authMessageBus = authMessageBus;
  userStreamController = StreamController<User>();
}
